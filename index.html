var map = L.map('map'), realtime = L.realtime({ url: 'https://wanderdrone.appspot.com/', crossOrigin: true, type: 'json' }, { interval: 3 * 1000 }).addTo(map); L.tileLayer('http://{s}.tile.osm.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(map); realtime.on('update', function() { map.fitBounds(realtime.getBounds(), {maxZoom: 3}); });
